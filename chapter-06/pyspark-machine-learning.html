Tuning Random Forest#

forest = RandomForestRegressor()

paramGrid = ParamGridBuilder()\
    .addGrid(forest.numTrees, [32, 64])\
    .addGrid(forest.minInstancesPerNode, [5])\
    .addGrid(forest.featureSubsetStrategy, ['onethird', 'sqrt', 'log2'])\

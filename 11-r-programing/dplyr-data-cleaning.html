Using z-score#

outliers_zscore <- function(array, z){
    array = as.vector(array)
    mean = array %>% mean
    std = array %>% sd
    lower = mean - z*std
    upper = mean + z*std
    array[(array < lower) | (array > upper)] = NA
    return(array)
    
}

Using interquartile range#

outliers_iqr <- function(array){
    array = as.vector(array)
    Q1 =  quantile(array, 0.25)
    Q3 =  quantile(array, 0.75)
    IQR = Q3 - Q1
    lower = Q1 - 1.5*IQR
    upper = Q3 + 1.5*IQR
    array[(array < lower) | (array > upper)] = NA
    return(array)
}
